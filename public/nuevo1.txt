// 🏁 LAP EN TIEMPO REAL
socket.on("lap", lap => {
  if (!latestLiveData) {
    latestLiveData = { classification: [], bestLapTime: null };
  }

  // 🔑 Resolver participante por transponder
  const participant = participantsCache.find(p => {
    const activeTR = p.kart_id
      ? p.kart_transponder
      : p.driver_transponder;
    return activeTR && String(activeTR) === String(lap.transponder);
  });

  if (!participant) {
    console.warn("⚠ Lap recibido con transponder no asignado:", lap.transponder);
    return;
  }

  const participantId = participant.id;

  let row = latestLiveData.classification.find(
    r => r.participant_id === participantId
  );

  if (!row) {
    row = {
      participant_id: participantId,
      lapcount: 0,
      lastTime: null,
      best: null,
      time: 0
    };
    latestLiveData.classification.push(row);
  }

  row.lapcount = lap.lap_number;
  row.lastTime = lap.lap_time_ms;
  row.best = row.best == null
    ? lap.lap_time_ms
    : Math.min(row.best, lap.lap_time_ms);

  row.time += lap.lap_time_ms;

  if (
    latestLiveData.bestLapTime == null ||
    lap.lap_time_ms < latestLiveData.bestLapTime
  ) {
    latestLiveData.bestLapTime = lap.lap_time_ms;
  }

  renderLiveFromLatest();
});

  function formatMs(ms) {
    if (ms == null || isNaN(ms)) return "-";
    let total = Math.max(0, ms);
    let h = Math.floor(total / 3600000);
    total %= 3600000;
    let m = Math.floor(total / 60000);
    total %= 60000;
    let s = Math.floor(total / 1000);
    let ms2 = total % 1000;
    return String(h).padStart(2,"0")+":"+String(m).padStart(2,"0")+":"+String(s).padStart(2,"0")+"."+String(ms2).padStart(3,"0");
  }

  function formatLapMs(ms) {
    if (ms == null || isNaN(ms)) return "-";

    let m = Math.floor(total / 60000);
    total %= 60000;
    let s = Math.floor(total / 1000);
    let ms2 = total % 1000;

async function saveGridEdit() {
  if (!editingGridEntry) return;

  const payload = {};

  // ===== GRID =====
  const gridInput = document.getElementById("edit-grid").value;
  const gridPos = Number(gridInput);
  if (!isNaN(gridPos) && gridPos > 0 && gridPos !== Number(editingGridEntry.grid_position)) {
    payload.grid_position = gridPos;
  }

  // ===== KART =====
  const kartValue = document.getElementById("edit-kart").value;
  let kartChanged = false;
  let newKartId = null;

  if (kartValue !== "") {
    newKartId = Number(kartValue);
    if (!isNaN(newKartId) && newKartId !== Number(editingGridEntry.kart_id)) {
      kartChanged = true;
    }
  }

  // 🚨 NADA QUE GUARDAR
  if (Object.keys(payload).length === 0 && !kartChanged) {
    alert("No hay cambios para guardar");
    return;
  }

  try {
    // ===== GUARDAR GRID (si cambió) =====
    if (Object.keys(payload).length > 0) {
      const res = await fetch(
        `${API}/race-control/entries/${editingGridEntry.id}`,
        {
          method: "PUT",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(payload)
        }
      );

      if (!res.ok) {
        alert(await res.text());
        return;
      }
    }

    // ===== CAMBIAR KART (si cambió) =====
    if (kartChanged) {
      const resKart = await fetch(
        `${API}/race-control/entries/${editingGridEntry.id}/change-kart`,
        {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ new_kart_id: newKartId })
        }
      );

      if (!resKart.ok) {
        alert(await resKart.text());
        return;
      }
    }

    closeGridEditor();
    await loadParticipants();
    await loadLiveExtended();

  } catch (err) {
    console.error("❌ Error guardando edición de parrilla:", err);
    alert("Error guardando cambios");
  }
}

function updateTimingTableExtended(rows, meta) {
  const body = document.getElementById("timing-body");
  const leaderBadge = document.getElementById("badge-leader");
  const pillLeaderLaps = document.getElementById("pill-leader-laps");
  const pillBestLap = document.getElementById("pill-best-lap");

  if (!rows || !rows.length) {
    body.innerHTML = '<tr><td colspan="14">Sin datos</td></tr>';
    leaderBadge.textContent = "Líder: V 0";
    pillLeaderLaps.textContent = "0";
    pillBestLap.textContent = "-";
    return;
  }

  const bestSession = meta && meta.bestLapTime != null ? meta.bestLapTime : null;
  const newPrevLapCounts = {};

  const leaderByLaps = rows.reduce(
    (acc, r) => (!acc || (r.lapcount || 0) > (acc.lapcount || 0)) ? r : acc,
    null
  );

  if (leaderByLaps) {
    leaderBadge.textContent = "Líder: V " + (leaderByLaps.lapcount || 0);
    pillLeaderLaps.textContent = leaderByLaps.lapcount || 0;
  } else {
    leaderBadge.textContent = "Líder: V 0";
    pillLeaderLaps.textContent = "0";
  }

  pillBestLap.textContent =
    bestSession != null ? formatLapMs(bestSession) : "-";

  body.innerHTML = rows.map(r => {
    const participantId = r.participant_id;
    const prevLap = prevLapCounts[participantId] || 0;
    const hasNewLap = (r.lapcount || 0) > prevLap;

    newPrevLapCounts[participantId] = r.lapcount || 0;

    const finishDotClass = hasNewLap ? "finish-dot active" : "finish-dot";
    const cat = r.category || autoCategory(r.weight);

    return `
      <tr class="${r.pos === 1 ? "leader" : ""}"
          ondblclick="openLapsPanel(${participantId})">

        <td class="finish-cell">
          <span class="${finishDotClass}"></span>
        </td>

        <td>${r.pos || "-"}</td>
        <td>${safeText(r.racerNumber)}</td>
        <td>${safeText(r.racerName)}</td>
        <td>${cat || "-"}</td>
        <td>${r.lapcount || 0}</td>

        <td>${r.lastTime != null ? formatLapMs(r.lastTime) : "-"}</td>
        <td>${r.best != null ? formatLapMs(r.best) : "-"}</td>

        <td class="time-cell">${r.time != null ? formatMs(r.time) : "-"}</td>
        <td>${safeText(r.gap)}</td>
        <td>${safeText(r.diff)}</td>
        <td>${r.mean != null ? formatLapMs(r.mean) : "-"}</td>
        <td>${r.consistency != null ? r.consistency.toFixed(1) + "%" : "-"}</td>
        <td>${safeText(r.racerTransponder)}</td>

      </tr>
    `;
  }).join("");

  prevLapCounts = newPrevLapCounts;
}

  function maybeAnnounceLap(r) {
    if (!voiceEnabled) return;
    if (!("speechSynthesis" in window)) return;

    const number = safeText(r.racerNumber);
    const name = safeText(r.racerName);
    const lap = r.lapcount || 0;

    let text = "";
    if (voiceMode === "kart") text = `Kart ${number}, vuelta ${lap}`;
    else if (voiceMode === "driver") text = `Piloto ${name}, vuelta ${lap}`;
    else text = `Kart ${number}, piloto ${name}, vuelta ${lap}`;

    const utter = new SpeechSynthesisUtterance(text);
    utter.lang = "es-ES";
    utter.rate = voiceRate;
    utter.volume = voiceVolume;
    window.speechSynthesis.speak(utter);
  }

  async function startRace() {
    if (!currentSession) return;
    if (!confirm("¿Iniciar la sesión? El cronómetro se pondrá a cero.")) return;

    const res = await fetch(`${API}/sessions/${currentSession.id}/start`, { method: "POST" });
    const data = await res.json().catch(() => ({}));
    if (data.reset_clock) raceStartTime = Date.now();
    await selectSession(currentSession.id);
  }

  async function pauseRace() {
    if (!currentSession) return;

    // Intentamos endpoints típicos: /pause y /resume
    if (currentSession.status === "running") {
      if (!confirm("¿Pausar la sesión?")) return;
      await fetch(`${API}/sessions/${currentSession.id}/pause`, { method: "POST" }).catch(()=>{});
      stopRaceTimer();
      await selectSession(currentSession.id);
      return;
    }

    if (currentSession.status === "paused") {
      await fetch(`${API}/sessions/${currentSession.id}/resume`, { method: "POST" }).catch(()=>{});
      // si el backend no tiene /resume, selectSession igualmente refresca
      await selectSession(currentSession.id);
    }
  }

    if (!currentSession) return;
    if (!confirm("⚠ ¿Finalizar la sesión?")) return;

    await fetch(`${API}/sessions/${currentSession.id}/finish`, { method: "POST" });

  // ========= CRONÓMETRO LOCAL =========
  function startRaceTimer() {
    if (!currentSession) return;

    const baseSec = Number(currentSession.elapsed_seconds || 0);
    const isRunning = currentSession.status === "running";
    let elapsedMs = baseSec * 1000;

    if (isRunning && currentSession.started_at) {
      const startedAtMs = new Date(currentSession.started_at).getTime();
      const nowMs = Date.now();
      if (!isNaN(startedAtMs)) elapsedMs += Math.max(0, nowMs - startedAtMs);
    }

    raceStartTime = Date.now() - elapsedMs;

    stopRaceTimer();
    raceTimerInterval = setInterval(updateRaceTimer, 100);
    updateRaceTimer();
  }

  function stopRaceTimer() {
    if (raceTimerInterval) clearInterval(raceTimerInterval);
    raceTimerInterval = null;
  }

  function updateRaceTimer() {
    if (!raceStartTime) return;
    const diff = Date.now() - raceStartTime;
    document.getElementById("race-timer-header").textContent = formatMs(diff);
  }

    if (!res.ok) return alert(await res.text());

    document.getElementById("edit-lap-menu").style.display = "none";
    await loadLiveExtended();
    await loadLapsForParticipant(pid);
  }

  // ========= HISTORIAL VUELTAS =========
  async function openLapsPanel(pid) {
    document.getElementById("laps-panel").style.display = "block";
    await loadLapsForParticipant(pid);
  }

  function closeLapsPanel() {
    document.getElementById("laps-panel").style.display = "none";
  }

  async function loadLapsForParticipant(pid) {
    if (!currentSession) return;
    const res = await fetch(`${API}/sessions/${currentSession.id}/participants/${pid}/laps`);
    if (!res.ok) return;

    const laps = await res.json();
    const p = participantsCache.find(x => Number(x.id) === Number(pid));
    const header = document.getElementById("laps-panel-header");
    header.textContent = p ? `Piloto: ${p.driver_name || "-"} | Kart: ${p.kart_number != null ? "#" + p.kart_number : "-"}` : "Piloto / Kart";

    const body = document.getElementById("laps-panel-body");
    if (!laps.length) {
      body.innerHTML = '<tr><td colspan="3" class="muted">Sin vueltas</td></tr>';
      return;
    }

    body.innerHTML = laps.map(l => `
      <tr>
        <td>${l.lap_number}</td>
        <td>${formatLapMs(l.lap_time_ms)}</td>
        <td>
          <button class="btn" style="padding:0.1rem 0.4rem;font-size:0.75rem;" onclick="editLapTime(${pid}, ${l.id}, ${l.lap_time_ms})">✏</button>
          <button class="btn" style="padding:0.1rem 0.4rem;font-size:0.75rem;background:#ef4444;border-color:#ef4444;" onclick="deleteLap(${pid}, ${l.id})">🗑</button>
        </td>
      </tr>`).join("");
  }

  async function deleteLap(pid, lapId) {
    if (!currentSession) return;
    if (!confirm("¿Borrar esta vuelta?")) return;

    const res = await fetch(`${API}/sessions/${currentSession.id}/participants/${pid}/laps/${lapId}`, { method:"DELETE" });
    if (!res.ok) return alert(await res.text());

    await loadLiveExtended();
    await loadLapsForParticipant(pid);
  }

  async function editLapTime(pid, lapId, ms) {
    if (!currentSession) return;
    const newVal = prompt("Nuevo tiempo (ms):", ms);
    if (newVal === null) return;

    const n = Number(newVal);
    if (isNaN(n) || n <= 0) return alert("Valor inválido");

    const res = await fetch(`${API}/sessions/${currentSession.id}/participants/${pid}/laps/${lapId}`, {
      method:"PUT",
      headers:{"Content-Type":"application/json"},
      body:JSON.stringify({ lap_time_ms:n })
    });
    if (!res.ok) return alert(await res.text());

    await loadLiveExtended();
    await loadLapsForParticipant(pid);
  }

    document.getElementById("min-lap-input").value = saved;
    minLapMs = saved;
  })();
